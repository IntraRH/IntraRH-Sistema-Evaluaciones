CREATE TABLE irh_proyectos (id bigint generated always as identity primary key, data_json jsonb, created_at timestamp default now());
CREATE TABLE irh_respuestas (id bigint generated always as identity primary key, data_json jsonb, created_at timestamp default now());
CREATE TABLE irh_participantes (id bigint generated always as identity primary key, data_json jsonb, created_at timestamp default now());
CREATE TABLE irh_proj_users (id bigint generated always as identity primary key, data_json jsonb, created_at timestamp default now());
CREATE TABLE irh_links_store (id bigint generated always as identity primary key, links_json jsonb, updated_at text);
CREATE TABLE irh_colaboradores (id bigint generated always as identity primary key, nombre text, correo text, empresa text, dept text, cargo text, created_at timestamp default now());
CREATE TABLE irh_config (id bigint generated always as identity primary key, config_json jsonb, created_at timestamp default now());
-- Permisos anon para respuestas y participantes
ALTER TABLE irh_respuestas ENABLE ROW LEVEL SECURITY;
ALTER TABLE irh_participantes ENABLE ROW LEVEL SECURITY;
CREATE POLICY "anon insert" ON irh_respuestas FOR INSERT TO anon WITH CHECK (true);
CREATE POLICY "anon insert" ON irh_participantes FOR INSERT TO anon WITH CHECK (true);
