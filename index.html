Ejecuta en el editor SQL de Supabase si las tablas no existen.

-- Tabla de usuarios del sistema
CREATE TABLE irh_users (
  id bigint generated always as identity primary key,
  nombre text,
  correo text unique,
  password_hash text,
  rol text default 'viewer' check (rol in ('owner','admin','viewer')),
  estado text default 'activo' check (estado in ('activo','inactivo')),
  auth_uid text unique,
  creado timestamp default now(),
  created_at timestamp default now()
);
ALTER TABLE irh_users ENABLE ROW LEVEL SECURITY;
CREATE POLICY "auth_all_users" ON irh_users FOR ALL TO authenticated USING (true) WITH CHECK (true);
CREATE INDEX idx_irh_users_correo ON irh_users(correo);
CREATE INDEX idx_irh_users_auth_uid ON irh_users(auth_uid);

-- Resto de tablas
CREATE TABLE irh_proyectos (id bigint generated always as identity primary key, data_json jsonb, created_at timestamp default now());
CREATE TABLE irh_respuestas (id bigint generated always as identity primary key, data_json jsonb, created_at timestamp default now());
CREATE TABLE irh_participantes (id bigint generated always as identity primary key, data_json jsonb, created_at timestamp default now());
CREATE TABLE irh_proj_users (id bigint generated always as identity primary key, data_json jsonb, created_at timestamp default now());
CREATE TABLE irh_links_store (id bigint generated always as identity primary key, links_json jsonb, updated_at text);
CREATE TABLE irh_colaboradores (id bigint generated always as identity primary key, nombre text, correo text, empresa text, dept text, cargo text, created_at timestamp default now());
CREATE TABLE irh_config (id bigint generated always as identity primary key, config_json jsonb, created_at timestamp default now());
-- Permisos anon para respuestas y participantes
ALTER TABLE irh_respuestas ENABLE ROW LEVEL SECURITY;
ALTER TABLE irh_participantes ENABLE ROW LEVEL SECURITY;
-- Permisos de lectura para anon (necesario para cargar proyectos en modo participante)
CREATE POLICY "anon select proyectos" ON irh_proyectos FOR SELECT TO anon USING (true);
CREATE POLICY "anon select links" ON irh_links_store FOR SELECT TO anon USING (true);
CREATE POLICY "anon select config" ON irh_config FOR SELECT TO anon USING (true);
CREATE POLICY "anon select participantes" ON irh_participantes FOR SELECT TO anon USING (true);
-- Permisos de escritura para anon (guardar respuestas de participantes)
CREATE POLICY "anon insert respuestas" ON irh_respuestas FOR INSERT TO anon WITH CHECK (true);
CREATE POLICY "anon select respuestas" ON irh_respuestas FOR SELECT TO anon USING (true);
CREATE POLICY "anon update respuestas" ON irh_respuestas FOR UPDATE TO anon USING (true) WITH CHECK (true);
CREATE POLICY "anon insert participantes" ON irh_participantes FOR INSERT TO anon WITH CHECK (true);
CREATE POLICY "anon update participantes" ON irh_participantes FOR UPDATE TO anon USING (true) WITH CHECK (true);
-- Habilitar RLS en todas las tablas
ALTER TABLE irh_proyectos ENABLE ROW LEVEL SECURITY;
ALTER TABLE irh_respuestas ENABLE ROW LEVEL SECURITY;
ALTER TABLE irh_participantes ENABLE ROW LEVEL SECURITY;
ALTER TABLE irh_links_store ENABLE ROW LEVEL SECURITY;
ALTER TABLE irh_config ENABLE ROW LEVEL SECURITY;
